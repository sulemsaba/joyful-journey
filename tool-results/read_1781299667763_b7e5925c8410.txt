   108→function createFallbackPage<TContent>(
   109→  slug: string,
   110→  title: string,
   111→  content: TContent,
   112→  metaTitle?: string,
   113→  metaDescription?: string
   114→): PageRecord<TContent> {

   116→    id: FALLBACK_RECORD_ID,
   117→    title,
   118→    slug,
   119→    content,
   120→    metaTitle,
   121→    metaDescription,
   122→    isPublished: true,
   123→    createdAt: FALLBACK_TIMESTAMP,
   124→    updatedAt: FALLBACK_TIMESTAMP,
   125→  };
   126→}
